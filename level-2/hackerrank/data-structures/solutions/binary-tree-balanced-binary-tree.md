def preOrder(root):
    res = []
    stk = []
    stk.append(root)
    while stk:
        curr = stk.pop()
        if curr.right != None:
            stk.append(curr.right)
        if curr.left != None:
            stk.append(curr.left)
        res.append(curr.info)
    print(*res)

void preOrder(Node* root) {

## Tree: Height of a Binary Tree

```python
def height(root):
    res = 0
    stk = []
    height = []
    stk.append(root)
    height.append(0)
    while stk:
        curr = stk.pop()
        curr_height = height.pop()
        if curr.left != None:
            stk.append(curr.left)
            height.append(curr_height+1)
        if curr.right != None:
            stk.append(curr.right)
            height.append(curr_height+1)
        res = max(res, curr_height)
    return res
```

def insert(self, val):
    if self.root == None:
        self.root = Node(val)
        return
    curr = self.root
    while True:
        if val < curr.info:
            if curr.left:
                curr = curr.left
            else:
                curr.left = Node(val)
                break
        elif val > curr.info:
            if curr.right:
                curr = curr.right
            else:
                curr.right = Node(val)
                break
        else:
            break

    <summary><h5>Python Solution</h5></summary>

```python
def lca(root, v1, v2):
    if root == None:
        return None
    if v1 > v2:
        v1, v2 = v2, v1
    while root.info < v1 or root.info > v2:
        if root.info < v1:
            root = root.right
        elif root.info > v2:
            root = root.left
    return root
```

def decodeHuff(root, s):
    decoded_str = ''
    curr = root
    for c in s:
        if c == '0' and curr.left:
            curr = curr.left
        elif c == '1' and curr.right:
            curr = curr.right
        if not curr.left and not curr.right:
            decoded_str += curr.data
            curr = root
    print(decoded_str)

class node:
    def __init__(self, val=0):
        self.val = val
        self.left = None

    if x == None:
        return 0

    if x == None:
        return 0

def rightRotate(y):
    x = y.left
    T2 = x.right

    x.right = y
    y.left = T2

    y.ht = max(get_height(y.left), get_height(y.right)) + 1
    x.ht = max(get_height(x.left), get_height(x.right)) + 1

    return x

    y = x.right
    T2 = y.left

    y.left = x
    x.right = T2

    y.ht = max(get_height(y.left), get_height(y.right)) + 1

    return y

    if curr == None:
        return node(val)
    if val < curr.val:
        curr.left = insert(curr.left, val)
    elif val > curr.val:
        curr.right = insert(curr.right, val)
    curr.ht = 1 + max(get_height(curr.left), get_height(curr.right))

        if val < curr.left.val:
            return rightRotate(curr)
        else:
            curr.left = leftRotate(curr.left)
            return rightRotate(curr)
