    res = []

def kangaroo(x1, v1, x2, v2):
    if x1 == x2:
        return "YES"
    if v1 == v2:
        return "NO"
    xdiff = x1 - x2
    vdiff = v2 - v1
    if (xdiff < 0 and vdiff < 0) or (xdiff > 0 and vdiff > 0):

        mod2 = vdiff % xdiff

    res = 0

    <summary>Python Solution</summary>

```python
def minimumDistances(a):
    res = 2e9
    for i in range(len(a)):
        for j in range(i+1, len(a)):
            if a[i] == a[j] and res > j-i:
                res = j-i
    if res == 2e9:
        return -1
    else:
        return res
```
